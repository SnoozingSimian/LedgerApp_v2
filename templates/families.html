/** * Update UI based on whether it's the Personal family * Hide invite button, delete button, and pending invites section, show notice instead */ function updateFamilyUIState(familyName) { const isPersonal = familyName && familyName.toLowerCase() === "personal"; const inviteBtn = document.getElementById("invite-member-btn"); const deleteBtn = document.getElementById("delete-family-btn"); const pendingInvitesSection = document.getElementById("pending-invites-details"); const personalNotice = document.getElementById("personal-family-notice"); if (isPersonal) { // Hide invite button for Personal family inviteBtn.style.display = "none"; // Hide delete button for Personal family deleteBtn.style.display = "none"; // Hide pending invites section for Personal family pendingInvitesSection.style.display = "none"; // Show Personal family notice personalNotice.classList.remove("hidden"); } else { // Show invite button for other families inviteBtn.style.display = "inline"; // Show delete button for other families deleteBtn.style.display = "block"; // Show pending invites section for other families pendingInvitesSection.style.display = "block"; // Hide Personal family notice personalNotice.classList.add("hidden"); } }
